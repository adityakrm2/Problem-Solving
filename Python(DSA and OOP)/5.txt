#Hard Level - Order of characters randomised:
#e.g. 4 letter, 2 symbol, 2 number = g^2jk8&P
c=[]
for i in range(0,nr_letters):
  b = random.randint(1,len(letters))
  c+=letters[b-1]
for i in range(0,nr_numbers):
  d= random.choice(numbers)
  c+=d
for i in range(0,nr_symbols):
  d= random.randint(1,len(symbols))
  c+=symbols[d-1]
# print(c)
random.shuffle(c)
# print(c)
password=""
for i in c:
  password+=i
print(f"Your Generated pass is {password}")
